\providecommand{\hccpInfTens}{%
  \begin{prooftree*}
    \AXC{$\seq[{ P }]{
        \mathcal{G} \hsep \ty{\Gamma},
        \tmty{x}{B} }$}
    \SYM{\tens}
    \UIC{$\seq[{ \piSend{x}{y}{P} }]{
        \mathcal{G} \hsep \ty{\Gamma},
        \tmty{x}{A \tens B}, \tmty{y}{A^\bot}
      }$}
  \end{prooftree*}}
